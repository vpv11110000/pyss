440  def run(self):
441 
442  self[START_TIME] = self._firstTx()
443  currentTime = self[START_TIME]
444  # ВАЖНО использовать, есть компоненты обращающиеся к components.curTime
445  # напрямую
446  self.setCurrentTime(currentTime)
447  self[START_TIME] = currentTime
448  while True:
449  if self[OPTIONS].logTransactTrace:
450  logger.info("### Time=%s " % str(currentTime) + 40 * "#")
451  # else:
452  # logger.display("\r>>> Time=%s"%str(currentTime))
453  try:
454  for s in self[SEGMENTS]:
455  self.setCurrentSegment(s)
456  if self[OPTIONS].printSegmentLabel and s[LABEL]:
457  logger.info("--- Segment: %s " % str(s[LABEL]) + 20 * "-")
458  s.handle(currentTime)
459  except pyssobject.TerminationCounterIsEmpty:
460  pass
461  if self.terminateCounter.isEmpty():
462  logger.info("\nStop by terminateCounter is Zero")
463  break
464  nextTime = self.findMinTime()
465  if nextTime <= currentTime or not nextTime:
466  logger.info("\nextTime<=currentTime or not nextTime")
467  break
468  if self[TERMINATION_CRITERIA] is not None and self[TERMINATION_CRITERIA]():
469  break
470  if self[MAX_TIME_STR]:
471  if nextTime > self[MAX_TIME_STR]:
472  logger.info("\nStop by Max Time")
473  break
474  oldTime = currentTime
475  currentTime = nextTime
476  # ВАЖНО использовать, есть компоненты обращающиеся к components.curTime
477  # напрямую
478  self.setCurrentTime(currentTime)
479  self.fireHandlerOnStateChange(oldState=oldTime)
480  self[END_TIME] = currentTime
