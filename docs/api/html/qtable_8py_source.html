00024

00028

00032     limitUpFirst=None - верхний предел первого частотного интервала
00033     widthInt=None - ширина частотного интервала
00034     countInt=None - число частотных интервалов (положительное целое число)
00035 
00036 Владельцем QTable является модель.
00037

00039

00066         #
00067         # limitUpFirst is None - вся область определения, счётчик в self[TOTAL]
00068         #
00069         # limitUpFirst is not None and countInt == 0 - область значений разделена

00114                 countInt -= 1
